<body>

    <h1>Choose Your Career Path</h1>

    <div id="careerChoices">

        <div class="option" onclick="selectCareer('Tech')">Tech</div>

        <div class="option" onclick="selectCareer('Medicine')">Medicine</div>

        <div class="option" onclick="selectCareer('Business')">Business</div>

        <div class="option" onclick="selectCareer('Arts')">Arts</div>

        <div class="option" onclick="selectCareer('Fashion')">Fashion</div>

    </div>

    

    <div id="educationChoices" class="hidden">

        <h2>Choose Your Education Path</h2>

        <div class="option" onclick="selectEducation('College Degree')">College Degree</div>

        <div class="option" onclick="selectEducation('Self-Taught')">Self-Taught</div>

        <div class="option" onclick="selectEducation('Trade School')">Trade School</div>

        <div class="option" onclick="selectEducation('Internship')">Internship/Work Experience</div>

    </div>

    

    <div id="result" class="hidden">

        <h2>Your Career Outcome</h2>

        <p id="outcomeText"></p>

    </div>

    

    <script>

        let career = "";

        let education = "";


        function selectCareer(choice) {

            career = choice;

            console.log("Selected Career:", career);  // Debugging line

            document.getElementById("careerChoices").classList.add("hidden");

            document.getElementById("educationChoices").classList.remove("hidden");

        }


        function selectEducation(choice) {

            education = choice;

            console.log("Selected Education:", education);  // Debugging line

            document.getElementById("educationChoices").classList.add("hidden");

            document.getElementById("result").classList.remove("hidden");

            generateOutcome();

        }


        function generateOutcome() {

            let outcome = "";

            

            if (education === "College Degree") {

                outcome = `You pursued a ${education} in ${career}. You have job security but also student debt.`;

            } else if (education === "Self-Taught") {

                outcome = `You chose to be ${education} in ${career}. Your skills are strong, but breaking into the industry is harder without credentials.`;

            } else if (education === "Trade School") {

                outcome = `You attended a ${education} and gained hands-on experience in ${career}. You find work faster with specialized skills.`;

            } else if (education === "Internship") {

                outcome = `You took the ${education} route in ${career}. Networking helped you secure opportunities, but gaining full-time roles takes time.`;

            }

            

            document.getElementById("outcomeText").innerText = outcome;

        }

    </script>
